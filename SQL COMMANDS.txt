This Project was done on Oracle Database Express Edition.
SQL COMMANDS:


CREATE SEQUENCE   DISTRICTSEQUENCE  MINVALUE 1 MAXVALUE 65 INCREMENT BY 1 START WITH 1 NOCACHE  ORDER  CYCLE;
CREATE SEQUENCE   SEATSEQUENCE  MINVALUE 1 MAXVALUE 350 INCREMENT BY 1 START WITH 1 NOCACHE  ORDER  CYCLE;
CREATE SEQUENCE   TEAMSEQUENCE  MINVALUE 1 MAXVALUE 50 INCREMENT BY 1 START WITH 1 NOCACHE  ORDER  CYCLE;
CREATE SEQUENCE   CANDIDATESEQUENCE  MINVALUE 1 MAXVALUE 10000 INCREMENT BY 1 START WITH 1 NOCACHE  ORDER  CYCLE;
CREATE SEQUENCE   CENTERSEQUENCE  MINVALUE 1 MAXVALUE 1000000 INCREMENT BY 1 START WITH 1 NOCACHE  ORDER  CYCLE;


CREATE TABLE  DISTRICTS 
   (	DISTRICTID NUMBER NOT NULL ENABLE, 
	DISTRICTNAME VARCHAR2(100) NOT NULL ENABLE, 
	 CONSTRAINT "DISTRICTS_PK" PRIMARY KEY (DISTRICTID) ENABLE
   );

CREATE OR REPLACE TRIGGER  "BI_DISTRICTS" 
  before insert on "DISTRICTS"               
  for each row  
begin   
    select "DISTRICTSEQUENCE".nextval into :NEW.DISTRICTID from dual; 
end;

ALTER TRIGGER  "BI_DISTRICTS" ENABLE;


CREATE TABLE  SEATS
   (	SEATID NUMBER NOT NULL ENABLE, 
	SEATNAME VARCHAR2(100) NOT NULL ENABLE, 
	DISTRICTID NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "SEATS_PK" PRIMARY KEY (SEATID) ENABLE, 
	 CONSTRAINT "SEATS_FK" FOREIGN KEY (DISTRICTID)
	  REFERENCES  "DISTRICTS" (DISTRICTID) ON DELETE CASCADE ENABLE
